def is_valid(pos, a):
    for i in range(1, pos):
        if a[i] == a[pos] or abs(a[i] - a[pos]) == abs(i - pos):
            return False
    return True

def n_queens(n):
    cnt = 0
    a = [0] * (n + 1)
    k = 1
    a[k] = 0
    while k != 0:
        a[k] += 1
        while a[k] <= n and not is_valid(k, a):
            a[k] += 1
        if a[k] <= n:
            if k == n:
                cnt += 1
                print_solution(a)
            else:
                k += 1
                a[k] = 0
        else:
            k -= 1
    return cnt
n = 4
count = n_queens(n)
print("\n")
print("Total solutions:", count)

        from itertools import permutations
        def travelling_salesman(graph, start):
            n = len(graph)
            vertices = set(range(n))
            vertices.remove(start)
            min_path = float('inf')
            for path in permutations(vertices):
                path = (start,) + path + (start,)
                path_weight = sum(graph[path[i]][path[i+1]] for i in range(n))
                min_path = min(min_path, path_weight)
            return min_path
        graph = [[0, 10, 15, 20], [10, 0, 35, 25],[15, 35, 0, 30], [20, 25, 30, 0]]
        start = 0
        print(travelling_salesman(graph, start))
        
        
    

    class Client:
        def __init__(self):
            self.order_list = {}
            self.amount_payable = 0
        def bill(self):
            menu = {"pizza": 20, "burger": 15, "pie": 10}
            print("---------Bill---------")
            print("|Food Name \t|Count \t|Price \t|")
            for food_name, count in self.order_list.items():
                print(f"|{food_name}\t\t|{count}\t|{menu[food_name]}\t|")
                print("------------")
            print(f"Total Amount due: {self.amount_payable}")
    def order(client):
        menu = {"pizza": 20, "burger": 15, "pie": 10}
        food_name = input("Enter the food item you want to order: ")
        count = int(input("Enter the count: "))
        if food_name in menu:
            client.order_list[food_name] = count
            client.amount_payable += menu[food_name] * count
        else:
            print("Please select from the menu.")
    def handler():
        client = Client()
        if input("Do you want to see the menu? Type y for Yes. ").lower() == "y":
            menu = {"pizza": 20, "burger": 15, "pie": 10}
            print("Today's menu:")
            for food_name, price in menu.items():
                print(f"{food_name} = {price}")
        while input("Type y to order. ").lower() == "y":
            order(client)
            while True:
                reply = input("Type y to order another food OR Type bill to get the bill. ").lower()
                if reply == "y":
                    order(client)
                elif reply == "bill":
                    client.bill()
                    return
    handler()
    
    

        def print_board(board):
        print(" " + board[0] + " | " + board[1] + " | " + board[2] + " ")
        print("-----------")
        print(" " + board[3] + " | " + board[4] + " | " + board[5] + " ")
        print("-----------")
        print(" " + board[6] + " | " + board[7] + " | " + board[8] + " ")
    def check_win(board, player):
        return ((board[0] == player and board[1] == player and board[2] == player) or
                (board[3] == player and board[4] == player and board[5] == player) or
                (board[6] == player and board[7] == player and board[8] == player) or
                (board[0] == player and board[3] == player and board[6] == player) or
                (board[1] == player and board[4] == player and board[7] == player) or
                (board[2] == player and board[5] == player and board[8] == player) or
                (board[0] == player and board[4] == player and board[8] == player) or
                (board[2] == player and board[4] == player and board[6] == player))
    def tic_tac_toe():
        board = [' '] * 9
        player = 'X'
        print_board(board)
        while True:
            move = input("It's " + player + "'s turn. Enter position (1-9): ")
            try:
                move = int(move) - 1
                if board[move] == ' ':
                    board[move] = player
                    print_board(board)
                    if check_win(board, player):
                        print("Congratulations, " + player + " won!")
                        return
                    player = 'O' if player == 'X' else 'X'
                else:
                    print("That position is already taken. Try again!")
            except ValueError:
                print("Invalid input. Try again!")
    tic_tac_toe()

        queue = deque([start])
        visited = set(start)
        while queue:
            node = queue.popleft()
            print(node, end=' ')
            for neighbor in graph[node]:
                if neighbor not in visited:
                    queue.append(neighbor)
                    visited.add(neighbor)
